export const FormToggle = ({ checked, disabled, focus, hover }) => {

      )}
      role="switch"
      tabindex="0"
      aria-checked={checked}

      <div
        className={clsx(

        )}
      >
        <div
          className={clsx(
            "utrecht-form-toggle__thumb",

            !checked && "utrecht-form-toggle__thumb--not-checked",
            disabled && "utrecht-form-toggle__thumb--disabled"
          )}
        ></div>
      </div>
    </div>
  );
